saner <value>
  (regains this amount of sanity per month)
sensedead <value>
  (can sense the number of corpses (of all types) in a square)
reformloc <value>
  (respawn terrain for immortals (-1 = anywhere, 1000=home))
hideanimals <value>
  (animals following this commander gains forest stealth)
mirror <value>
  (1=is small mirror, 2=is large mirror, 3=silver, 4=gold)
revertmirror <value>
  (revert to mirror when empty. 1=small mirror, 2=large mirror, 3=silver, 4=gold)
phantasm <value>
  (1=phantasmal warrior, 2=phantasmal animal, etc.)
releasephant <value>
  (mirror releases this type of phantasms (has phantasm = value))
releaserate <value>
  (bonus to the number of phantasms released)
troll
  (is a troll)
noplanecamo <value>
  (does not use fx_lookslike when in plane = value)
homeplane <value>
  (will return to this plane if slain outside it and be indep controlled again)
allrit <value>
  (knows all rituals of this type e.g. fx_necropow, -1=all powers)
displaced <value>
  (25% chance of missing per rank away from target)
riverdmg <value>
  (takes an value damage when passing a river)
iceprot <value>
  (+value armor in cold climate)
melt <value>
  (takes value dmg per month when not cold)
voidsanity <value>
  (reduces insanity gains (like armor))
planeshift <value>
  (can go to/from other plane, 1=hades, 2=inferno)
aipowcom1 <value>
  (unit wants to be led by com with this power at level 1+)
aipowcom2 <value>
  (unit wants to be led by com with this power at level 2+)
aipowcom3 <value>
  (unit wants to be led by com with this power at level 3+)
releasespell <value>
  (making an attack drains one mirror charge (fx_mirror))
fear <value>
  (causes fear: 1=fear, 2=dread, 3=terror)
goldcarrier <value>
  (spoils of war when defeating this monster)
ironcarrier <value>
  (spoils of war when defeating this monster)
rebate <value>
  (give rebate on rituals with fx_rebatefx??)
makeruin <value>
  (value = chance of turning current square into a ruin if possible)
meleeambush <value>
  (Will start in melee combat when ambushing)
blesspierce <value>
  (Just like res_pierce but only for bless effects)
poisoncloud <value>
  (spreads poison cloud, value = radius + 1)
mirrorammo <value>
  (default number of images for this type of mirror (def 20))
affres <value>
  (value = chance of not receiving battle affliction)
wander <value>
  (1=random & peaceful, 2=crusade, 3=inquisition/saint, 4=rider, 5=angel, 6=devil, 7=ghost, 8=settlements)
infwander <value>
  (only wander if owned by INFPLAYER)
hadeswander <value>
  (only wander if owned by DEATHPLAYER)
maxsinners <value>
  (Will gather sinners when owned by INFPLAYER (value=amount before returning))
forestheart <value>
  (takes control of nearby forests)
spread <value>
  (unit will position itself spread out among the other troops)
seduceaura <value>
  (attacker within this range might get seduced instead)
fireres <value>
  (fire resistance)
coldres <value>
  (cold resistance)
shockres <value>
  (shock resistance)
poisonres <value>
  (poison resistance)
coldheal <value>
  (heals value hp/month when it's cold)
growtime <value>
  (will grow into next monster after about value turns)
growoffs <value>
  (to become other than next monster)
localironbonus <value>
  (att% bonus to iron income in this square (only works in mines))
localgoldbonus <value>
  (att% bonus to gold income in this square (only works in mines))
neverturn <value>
  (never draw facing the other way)
aigoldrally <value>
  (Hint that AI should go to recruitment loc when having lots of gold)
thrallhunt <value>
  (can create thralls (vampire))
ainofollower
  (AI hint to always use this commander as a leader (not a subcommander))
putridexpl <value>
  (Putrefy cloud upon death (value magic dmg + disease(mr)))
aimaxshop <value>
  (ai won't buy more than this amount at the same place)
maxsum <value>
  (the maximum amount that can be summoned in a single battle)
farsight <value>
  (can see further)
changetemp <value>
  (makes square value degrees hotter)
setcreator <value>
  (get a fx_createdby ufx when created)
statue <value>
  (get name from creator)
minorstartaff <value>
  (value = chance)
landshape <value>
  (will take shape +value when on land)
watershape <value>
  (will take shape +value when in the sea)
likestoburn <value>
  (value% chance of wanting to burn a forest (wandering indep coms only))
reanimate <value>
  (reanimates automatically if possible, value = max amount / month)
spawnmon <value>
  (Spawns next monster. value/100 = monsters per turn)
spawnoffs <value>
  (To spawn something other than the next monster)
spawnmonaway <value>
  (Spawns next monster, but not on homeplane. value/100 = monsters per turn)
spellrange <value>
  (value = +range for battle spells)
spawn1d6mon <value>
  (Spawns 1d6 of next monster. value = chance)
spawn2d6mon <value>
  (Spawns 2d6 of next monster. value = chance)
diseasecloud <value>
  (spreads disease cloud, value = radius + 1)
mindexpl <value>
  (value = 1)
limitgold <value>
  (value = amount of gold per month, but max the value of the square)
limitiron <value>
  (value = amount of iron per month, but max the value of the square)
reformdestroy <value>
  (chance to destroy location on immortality respawn)
deployoutside <value>
  (setup this unit outside fort in battles)
hadesres <value>
  (resistance to the effects of hades)
stonebeing
  (monster is made of stone)
trample <dmg>
  (monster will get a trample attack with this damage value)
tramplexsize <bonus size>
  (monster will count as this number of squares larger regarding trampling)
nocombat
  (monster is a non-combatant, like a floating eye)
acidblood <dmg>
  (acid blood strikeback effect)
noeyes 
  (monster has no eyes and doesn't need eyes to see)
makecolony <terrain nbr>
  (has a chance of turning terrain into this terrain and then die)
